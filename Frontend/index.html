def hitung_luas(panjang, lebar):
    return panjang * lebar

# Hitung beberapa persegi panjang
ukuran = [(5,3), (8,4), (10,6)]

for p, l in ukuran:
    luas = hitung_luas(p, l)
    print(f"{p}×{l} = {luas} m²")
▶ Output
5×3 = 15 m²
8×4 = 32 m²
10×6 = 60 m²|
